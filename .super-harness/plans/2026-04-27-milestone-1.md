// src/PngToBody.js
import { decomp } from 'https://cdn.jsdelivr.net/npm/poly-decomp@0.3.0/build/decomp.min.js';

const { Bodies, Body, Common } = Matter;

// Set decomp for fromVertices
if (Common.setDecomp) Common.setDecomp(decomp);

export class PngToBody {

  /**
   * Extract contour from canvas alpha channel using marching squares
   */
  static extractContour(ctx, width, height, alphaThreshold = 128) {
    const imageData = ctx.getImageData(0, 0, width, height);
    const data = imageData.data;

    // Build binary grid
    const grid = [];
    for (let y = 0; y < height; y++) {
      grid[y] = [];
      for (let x = 0; x < width; x++) {
        const idx = (y * width + x) * 4;
        grid[y][x] = data[idx + 3] >= alphaThreshold ? 1 : 0;
      }
    }

    // Marching squares contour tracing
    const points = [];
    const visited = new Set();

    // Find starting point
    let startX = -1, startY = -1;
    for (let y = 0; y < height && startX === -1; y++) {
      for (let x = 0; x < width; x++) {
        if (grid[y][x] === 1) { startX = x; startY = y; break; }
      }
    }
    if (startX === -1) return points;

    // Simple boundary walk: collect all edge pixels
    for (let y = 0; y < height; y++) {
      for (let x = 0; x < width; x++) {
        if (grid[y][x] === 1) {
          const isEdge =
            x === 0 || y === 0 || x === width - 1 || y === height - 1 ||
            grid[y-1][x] === 0 || grid[y+1][x] === 0 ||
            grid[y][x-1] === 0 || grid[y][x+1] === 0;
          if (isEdge) points.push({ x, y });
        }
      }
    }

    // Sort points by angle from centroid to form ordered polygon
    if (points.length < 3) return points;
    const cx = points.reduce((s, p) => s + p.x, 0) / points.length;
    const cy = points.reduce((s, p) => s + p.y, 0) / points.length;
    points.sort((a, b) => Math.atan2(a.y - cy, a.x - cx) - Math.atan2(b.y - cy, b.x - cx));

    return points;
  }

  /**
   * Ramer-Douglas-Peucker polygon simplification
   */
  static simplifyPolygon(points, epsilon = 2) {
    if (points.length <= 3) return points;

    function rdp(pts, eps) {
      if (pts.length <= 2) return pts;

      let maxDist = 0, maxIdx = 0;
      const first = pts[0], last = pts[pts.length - 1];

      for (let i = 1; i < pts.length - 1; i++) {
        const dist = pointLineDistance(pts[i], first, last);
        if (dist > maxDist) { maxDist = dist; maxIdx = i; }
      }

      if (maxDist > eps) {
        const left = rdp(pts.slice(0, maxIdx + 1), eps);
        const right = rdp(pts.slice(maxIdx), eps);
        return left.slice(0, -1).concat(right);
      }
      return [first, last];
    }

    function pointLineDistance(p, a, b) {
      const dx = b.x - a.x, dy = b.y - a.y;
      const lenSq = dx * dx + dy * dy;
      if (lenSq === 0) return Math.hypot(p.x - a.x, p.y - a.y);
      const t = Math.max(0, Math.min(1, ((p.x - a.x) * dx + (p.y - a.y) * dy) / lenSq));
      return Math.hypot(p.x - (a.x + t * dx), p.y - (a.y + t * dy));
    }

    return rdp(points, epsilon);
  }

  /**
   * Create a matter.js body from canvas image data
   */
  static async createBody(ctx, width, height, options = {}) {
    const {
      x = 0, y = 0, scale = 1, density = 0.001,
      texturePath = null, fallbackWidth = null, fallbackHeight = null
    } = options;

    const contour = PngToBody.extractContour(ctx, width, height);

    let body;
    if (contour.length >= 3) {
      const simplified = PngToBody.simplifyPolygon(contour, 2);
      // Center vertices around origin
      const cx = simplified.reduce((s, p) => s + p.x, 0) / simplified.length;
      const cy = simplified.reduce((s, p) => s + p.y, 0) / simplified.length;
      const vertices = simplified.map(p => ({ x: (p.x - cx) * scale, y: (p.y - cy) * scale }));

      try {
        body = Bodies.fromVertices(x, y, vertices, { density }, true);
      } catch (e) {
        console.warn('Convex decomposition failed, falling back to bounding box', e);
      }
    }

    // Fallback: bounding box
    if (!body) {
      const fw = (fallbackWidth || width) * scale;
      const fh = (fallbackHeight || height) * scale;
      body = Bodies.rectangle(x, y, fw, fh, { density });
    }

    // Attach sprite texture
    if (texturePath) {
      body.render = body.render || {};
      body.render.sprite = { texture: texturePath, xScale: scale, yScale: scale };
    }

    return body;
  }
}

// src/PhysicsScene.js
const { Engine, Render, Runner, Bodies, Composite, Constraint } = Matter;

export class PhysicsScene {
  constructor(container, options = {}) {
    const { width = 800, height = 600 } = options;

    this.width = width;
    this.height = height;

    this.engine = Engine.create();
    this.engine.gravity.y = 1;

    this.render = Render.create({
      element: container,
      engine: this.engine,
      options: { width, height, wireframes: false, background: '#fafafa' }
    });

    this.runner = Runner.create();

    // Default ring: static circle at top-center
    this.ring = Bodies.circle(width / 2, 60, 40, { isStatic: true });
    Composite.add(this.engine.world, this.ring);
  }

  start() {
    Render.run(this.render);
    Runner.run(this.runner, this.engine);
  }

  stop() {
    Render.stop(this.render);
    Runner.stop(this.runner);
  }

  replaceRing(newRing) {
    const oldRing = this.ring;
    Composite.remove(this.engine.world, oldRing);
    this.ring = newRing;
    Composite.add(this.engine.world, newRing);
  }

  addBody(body) {
    Composite.add(this.engine.world, body);
  }

  removeBody(body) {
    Composite.remove(this.engine.world, body);
  }
}

// src/CharmManager.js
const { Constraint, Composite } = Matter;

export class CharmManager {
  constructor(scene) {
    this.scene = scene;
    this.charms = [];    // { body, constraint }
    this.maxCharms = 20;
  }

  addCharm(body, constraintLength = 80) {
    if (this.charms.length >= this.maxCharms) return false;

    const constraint = Constraint.create({
      bodyA: this.scene.ring,
      pointA: { x: 0, y: this.scene.ring.circleRadius },
      bodyB: body,
      pointB: { x: 0, y: -body.circleRadius || -15 },
      length: constraintLength,
      stiffness: 0.8
    });

    this.scene.addBody(body);
    this.scene.addBody(constraint);

    this.charms.push({ body, constraint });
    return true;
  }

  removeCharm(bodyId) {
    const idx = this.charms.findIndex(c => c.body.id === bodyId);
    if (idx === -1) return;

    const { body, constraint } = this.charms[idx];
    this.scene.removeBody(constraint);
    this.scene.removeBody(body);
    this.charms.splice(idx, 1);
  }

  get constraints() {
    return this.charms.map(c => c.constraint);
  }

  reconnectAllConstraints() {
    this.charms.forEach(({ body, constraint }) => {
      constraint.bodyA = this.scene.ring;
      constraint.pointA = { x: 0, y: this.scene.ring.circleRadius };
    });
  }
}

// src/InteractionManager.js
const { Mouse, MouseConstraint, Events, Composite } = Matter;

export class InteractionManager {
  constructor(scene, charmManager, canvas) {
    this.scene = scene;
    this.charmManager = charmManager;
    this.selectedCharm = null;
    this.mouseConstraint = null;
    this.draggingCharm = null;
    this.dragConstraint = null;

    this._setupMouseConstraint(canvas);
    this._setupKeyboard();
    this._setupCanvasClick(canvas);
    this.onDelete = null; // callback set by app.js
  }

  _setupMouseConstraint(canvas) {
    const mouse = Mouse.create(canvas);
    this.mouseConstraint = MouseConstraint.create(this.scene.engine, {
      mouse,
      constraint: { stiffness: 0.2, render: { visible: false } }
    });
    this.scene.addBody(this.mouseConstraint);

    // Track drag start/end for charm re-constraining
    Events.on(this.mouseConstraint, 'startdrag', (e) => {
      const body = e.body;
      if (this._isCharm(body)) {
        this.draggingCharm = body;
        this._removeCharmConstraint(body);
      }
    });

    Events.on(this.mouseConstraint, 'enddrag', (e) => {
      if (this.draggingCharm) {
        this._reconstrainCharm(this.draggingCharm);
        this.draggingCharm = null;
      }
    });
  }

  _setupKeyboard() {
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Delete' && this.selectedCharm) {
        this._deleteSelected();
      }
    });
  }

  _setupCanvasClick(canvas) {
    Events.on(this.mouseConstraint, 'mousedown', (e) => {
      const body = e.body;
      if (this._isCharm(body)) {
        this.selectCharm(body);
      } else {
        this.deselectCharm();
      }
    });
  }

  _isCharm(body) {
    return this.charmManager.charms.some(c => c.body.id === body.id);
  }

  selectCharm(body) {
    this.deselectCharm();
    this.selectedCharm = body;
    body.render.strokeStyle = '#ff6b6b';
    body.render.lineWidth = 3;
  }

  deselectCharm() {
    if (this.selectedCharm) {
      this.selectedCharm.render.strokeStyle = null;
      this.selectedCharm.render.lineWidth = 0;
      this.selectedCharm = null;
    }
  }

  _removeCharmConstraint(body) {
    const entry = this.charmManager.charms.find(c => c.body.id === body.id);
    if (entry && entry.constraint) {
      this.scene.removeBody(entry.constraint);
      entry.constraint = null;
    }
  }

  _reconstrainCharm(body) {
    const entry = this.charmManager.charms.find(c => c.body.id === body.id);
    if (!entry) return;

    const constraint = Matter.Constraint.create({
      bodyA: this.scene.ring,
      pointA: { x: 0, y: this.scene.ring.circleRadius },
      bodyB: body,
      pointB: { x: 0, y: -(body.circleRadius || 15) },
      length: 80,
      stiffness: 0.8
    });
    this.scene.addBody(constraint);
    entry.constraint = constraint;
  }

  _deleteSelected() {
    if (!this.selectedCharm) return;
    const id = this.selectedCharm.id;
    this.selectedCharm = null;
    this.charmManager.removeCharm(id);
    if (this.onDelete) this.onDelete(id);
  }

  deleteByToolbar() {
    this._deleteSelected();
  }
}

const MIME_TYPES = {
  '.html': 'text/html',
  '.css': 'text/css',

  '.json': 'application/json',
  '.png': 'image/png',
  '.jpg': 'image/jpeg',

function ensureManifest() {

  }
}

function serveStatic(filePath, res) {
  const ext = path.extname(filePath);
  fs.readFile(filePath, (err, data) => {
    if (err) {
      res.writeHead(404);
      res.end('Not found');
      return;
    }
    res.writeHead(200, { 'Content-Type': MIME_TYPES[ext] || 'application/octet-stream' });
    res.end(data);
  });
}

const server = http.createServer((req, res) => {
  const url = new URL(req.url, `http://localhost:${PORT}`);

  // CORS
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');
  if (req.method === 'OPTIONS') { res.writeHead(204); res.end(); return; }

  // GET /api/manifest
  if (req.method === 'GET' && url.pathname === '/api/manifest') {
    ensureManifest();
    serveStatic(MANIFEST_PATH, res);
    return;
  }

  // POST /api/upload
  if (req.method === 'POST' && url.pathname === '/api/upload') {
    const chunks = [];
    req.on('data', chunk => chunks.push(chunk));
    req.on('end', () => {
      const buffer = Buffer.concat(chunks);
      const category = url.searchParams.get('category') || 'charms';
      const filename = url.searchParams.get('filename');

      if (!filename || !filename.endsWith('.png')) {
        res.writeHead(400, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'Only PNG files allowed' }));
        return;
      }

      const dir = path.join(ASSETS_DIR, category);
      if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });

      const filePath = path.join(dir, filename);
      fs.writeFileSync(filePath, buffer);
      res.writeHead(200, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ success: true, path: `${category}/${filename}` }));
    });
    return;
  }

  // POST /api/manifest
  if (req.method === 'POST' && url.pathname === '/api/manifest') {
    const chunks = [];
    req.on('data', chunk => chunks.push(chunk));
    req.on('end', () => {
      try {
        const manifest = JSON.parse(Buffer.concat(chunks).toString());
        ensureManifest();
        fs.writeFileSync(MANIFEST_PATH, JSON.stringify(manifest, null, 2));
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ success: true }));
      } catch (e) {
        res.writeHead(400, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'Invalid JSON' }));
      }
    });
    return;
  }

  // Static files
  let filePath = path.join(__dirname, url.pathname === '/' ? 'index.html' : url.pathname);
  serveStatic(filePath, res);
});

server.listen(PORT, () => {
  ensureManifest();
  console.log(`Charm DIY dev server running at http://localhost:${PORT}`);

    const btn = zone.querySelector('.upload-btn');

    btn.addEventListener('click', () => fileInput.click());
    fileInput.addEventListener('change', (e) => this._handleFile(e.target.files[0]));

    zone.addEventListener('dragover', (e) => { e.preventDefault(); zone.classList.add('dragover'); });
    zone.addEventListener('dragleave', () => zone.classList.remove('dragover'));
    zone.addEventListener('drop', (e) => {
      e.preventDefault();
      zone.classList.remove('dragover');
      const file = e.dataTransfer.files[0];
      this._handleFile(file);
    });

    container.appendChild(zone);
    this.fileInput = fileInput;
  }

  async _handleFile(file) {
    if (!file || !file.name.endsWith('.png')) {
      this._showToast('仅支持 PNG 文件');
      return;
    }

    // Show preview dialog
    const url = URL.createObjectURL(file);
    const img = new Image();
    img.src = url;

    await new Promise(resolve => { img.onload = resolve; });

    const category = await this._showPreviewDialog(file.name, img);
    if (!category) return; // user cancelled

    // Upload to server
    const arrayBuffer = await file.arrayBuffer();
    const res = await fetch(
      `${this.serverUrl}/api/upload?category=${category}&filename=${file.name}`,
      { method: 'POST', body: arrayBuffer }
    );
    const result = await res.json();

    if (result.success) {
      // Update manifest
      const manifestRes = await fetch(`${this.serverUrl}/api/manifest`);
      const manifest = await manifestRes.json();
      const id = file.name.replace('.png', '');
      const defaultEntry = {
        id,
        file: result.path,
        scale: 1,
        ...(category === 'charms' ? { density: 0.001 } : { radius: Math.min(img.width, img.height) / 2 })
      };
      manifest[category].push(defaultEntry);
      await fetch(`${this.serverUrl}/api/manifest`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(manifest)
      });

      this._showToast(`${file.name} 已添加`);
      this.onUpload(category, defaultEntry);
    }
  }

  _showPreviewDialog(filename, img) {
    return new Promise((resolve) => {
      const dialog = document.createElement('div');
      dialog.className = 'upload-preview-dialog';
      dialog.innerHTML = `

            <label>分类:
              <select class="field-category">
                <option value="charms">饰物</option>
                <option value="rings">挂环</option>
              </select>
            </label>
          </div>
          <div class="dialog-actions">
            <button class="btn-cancel">取消</button>
            <button class="btn-confirm">确认</button>
          </div>
        </div>
      `;
      dialog.querySelector('.preview-image').appendChild(img);
      dialog.querySelector('.btn-cancel').addEventListener('click', () => {
        dialog.remove();
        resolve(null);
      });
      dialog.querySelector('.btn-confirm').addEventListener('click', () => {
        const category = dialog.querySelector('.field-category').value;
        dialog.remove();
        resolve(category);
      });
      document.body.appendChild(dialog);
    });
  }

    toast.className = 'toast';
    toast.textContent = message;

async function init() {
  const canvasContainer = document.getElementById('canvas-container');
  const sidebarEl = document.getElementById('sidebar');
  const deleteBtn = document.getElementById('delete-btn');
  const uploadContainer = document.getElementById('upload-container');

  // Init physics scene
  const scene = new PhysicsScene(canvasContainer, { width: 800, height: 600 });
  const charmManager = new CharmManager(scene);

  // Init interaction
  const canvas = canvasContainer.querySelector('canvas');
  const interaction = new InteractionManager(scene, charmManager, canvas);
  deleteBtn.addEventListener('click', () => interaction.deleteByToolbar());

  // Init sidebar
  const sidebarContent = document.getElementById('sidebar-content');
  const sidebar = new Sidebar(sidebarContent, {
    onRingSelect: (ringConfig) => replaceRing(ringConfig),
    onCharmSelect: (charmConfig) => addCharm(charmConfig)
  });
  await sidebar.loadManifest(SERVER_URL);

  // Init upload
  new UploadUI(uploadContainer, {
    serverUrl: SERVER_URL,
    onUpload: () => sidebar.loadManifest(SERVER_URL)
  });

  // Start physics
  scene.start();

  async function addCharm(config) {
    const img = new Image();
    img.src = `assets/${config.file}`;
    await new Promise(r => { img.onload = r; });

    const tempCanvas = document.createElement('canvas');
    tempCanvas.width = img.width;
    tempCanvas.height = img.height;
    const ctx = tempCanvas.getContext('2d');
    ctx.drawImage(img, 0, 0);

    const body = await PngToBody.createBody(ctx, img.width, img.height, {
      x: scene.ring.position.x + (Math.random() - 0.5) * 40,
      y: scene.ring.position.y + 100,
      scale: config.scale || 1,
      density: config.density || 0.001,
      texturePath: `assets/${config.file}`
    });

    charmManager.addCharm(body);
  }

  async function replaceRing(config) {
    const img = new Image();
    img.src = `assets/${config.file}`;
    await new Promise(r => { img.onload = r; });

    const newRing = Matter.Bodies.circle(scene.width / 2, 60, config.radius || 40, {
      isStatic: true,
      render: {
        sprite: { texture: `assets/${config.file}`, xScale: config.scale || 1, yScale: config.scale || 1 }
      }
    });

    scene.replaceRing(newRing);
    charmManager.reconnectAllConstraints();
  }
}

// Creates minimal ring-circle.png and star.png in assets/
const { createCanvas } = require('canvas');
const fs = require('fs');
const path = require('path');

function generateRingCircle() {
  const size = 80;
  const canvas = createCanvas(size, size);
  const ctx = canvas.getContext('2d');
  ctx.clearRect(0, 0, size, size);
  ctx.beginPath();
  ctx.arc(40, 40, 35, 0, Math.PI * 2);
  ctx.arc(40, 40, 25, 0, Math.PI * 2, true);
  ctx.fillStyle = '#c0a060';
  ctx.fill();
  return canvas;
}

function generateStar() {
  const size = 40;
  const canvas = createCanvas(size, size);
  const ctx = canvas.getContext('2d');
  ctx.clearRect(0, 0, size, size);
  ctx.beginPath();
  for (let i = 0; i < 5; i++) {
    const angle = (i * 4 * Math.PI / 5) - Math.PI / 2;
    const method = i === 0 ? 'moveTo' : 'lineTo';
    ctx[method](20 + 18 * Math.cos(angle), 20 + 18 * Math.sin(angle));
  }
  ctx.closePath();
  ctx.fillStyle = '#e94560';
  ctx.fill();
  return canvas;
}
